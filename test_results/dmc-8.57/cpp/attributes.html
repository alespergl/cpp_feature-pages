Reference:
    [N2761](http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2008/n2761.pdf)

Check with:
    #ifdef __cpp_attributes
    #if __has_feature(cxx_attributes)
*/

struct [[attr]] S {} s;

[[noreturn]] void f()
{
    throw "error";
}

    {
    case 1:
        a();
        [[fallthrough]];
    default:
        b();
    }
}
