struct S
{
    int a;
    const char* b;
    int c;
    int d = b[a];   // this should compile without errors
};

S ss = { 1, "asdf" };   // ss.d == 's'

int main()
{
    if(ss.d != 's') puts("ERROR: aggregate member initialization with equal-initializer doesn't work");
}
